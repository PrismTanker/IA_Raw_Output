def detect_zero_balance(operations):
    balance = 0
    for op in operations:
        if op['type'] == 'deposit':
            balance += op['amount']
        elif op['type'] == 'withdrawal':
            balance -= op['amount']
            if balance < 0:
                return True
    return False
