    encrypt('gf') returns 'kj'
    encrypt('et') returns 'ix'
    """
    encrypted = ''
    for char in s:
        if char.isalpha():
            encrypted += chr(ord(char) + 2 * 2)
        else:
            encrypted += char
    return encrypted
